# 固定随机种子，保证数据可复现
set.seed(2026)

# 样本量
n <- 200

# 1. 分类变量
gender <- sample(c("男", "女"), size = n, replace = TRUE) 
# 填充n个男女性别样本 
# replace 代表抽样是否重复，若是ture则代表可重复。
class  <- sample(c("一班", "二班", "三班"), size = n, replace = TRUE)

# 2. 连续变量
study_hour <- round(runif(n, 1.0, 6.0), 1)   # 每日学习时长
# runif 取1.0到6.0中的随机数，n个 
phone_hour <- round(runif(n, 1.0, 5.0), 1)   # 手机使用时长
sleep_hour <- round(rnorm(n, mean = 7, sd = 1), 1)  # 睡眠时长
# rnorm生成符合正态分布的数据，均值为7，方差为1
# 3. 成绩变量
mid_exam <- 45 + 4 * study_hour - 2.5 * phone_hour + rnorm(n, 0, 6)
mid_exam <- pmin(pmax(mid_exam, 0), 100)
# pmax取mid_exam和0中的较大值，pmin类似，达成数值截断的效果，不会出现小于0和大于100的情况

final_exam <- 40 + 3.2 * study_hour - 1.8 * phone_hour +
  0.5 * sleep_hour + 0.5 * mid_exam + rnorm(n, 0, 7)
final_exam <- pmin(pmax(final_exam, 0), 100)

# 4. 有序等级（不及格、及格、良好、优秀）
grade <- cut(final_exam,
             breaks = c(0, 60, 70, 85, 100),
             labels = c("不及格", "及格", "良好", "优秀"),
             ordered_result = TRUE)

# 5. 最终数据框
dat <- data.frame(
  gender,
  class,
  study_hour,
  phone_hour,
  sleep_hour,
  mid_exam,
  final_exam,
  grade
)

# 查看前几行
head(dat)

# 保存为 CSV 文件（Excel 也能打开）
write.csv(dat, "student_data.csv", row.names = FALSE)
# 查看当前工作目录（文件就存在这里）
getwd()
dat$scholarship <- ifelse(dat$final_exam >= 85, 1, 0)
